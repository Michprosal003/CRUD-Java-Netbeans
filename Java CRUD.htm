    void Modificar() {

       

    

    String nombre = txtNombre.getText();

    String apellido = txtApellido.getText();

    String codigo = txtCodigo.getText();

    String materia = txtMateria.getText();

    String nota = txtNota.getText();

    String jornada = txtJornada.getText();

    String estado = txtEstado.getText();

   

    try {

       

       

     if(nombre.equals("") || apellido.equals("") || codigo.equals("") || materia.equals("") || nota.equals("") || jornada.equals("") || estado.equals("")) {

     

                         JOptionPane.showMessageDialog(null,"Faltan ingresar datos");

                        

                         limpiarTabla();

 

        

     }

    

    

     else {

        

         String sql= " Update curso Set Id='"+idc+"', nombre='"+nombre+"', apellido='"+apellido+"', codigo='"+codigo+"', materia='"+materia+"', nota='"+nota+"',jornada='"+jornada+"', estado='"+estado+"' where id="+idc;

        

         st= conectar.createStatement();

        

         st.executeUpdate(sql);

        

          JOptionPane.showMessageDialog(null,"Datos de estudiante modificado");

         

          limpiarTabla();

 

        

        

     }
